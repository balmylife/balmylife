    <dict>
        <key>KeepAlive</key>
        <true/>
        <key>RunAtLoad</key>
        <true/>
        <key>Label</key>
        <string>local.Aria2</string>
        <key>ProgramArguments</key>
        <array>
            <string>/Users/felix/applications/aria2/Aria2.sh</string> 
        </array>
        <key>WorkingDirectory</key>
        <string>/Users/felix/Downloads</string>
    </dict>
</plist>
